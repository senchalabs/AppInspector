Singleton('Siesta.Test.Simulate.KeyCodes', {

    methods : {
        isNav : function (k) {
            var keys = this.keys

            return (k >= 33 && k <= 40) ||
                k == keys.RETURN ||
                k == keys.TAB ||
                k == keys.ESCAPE;
        },

        isSpecial : function (k) {

            return (k === this.keys.BACKSPACE) ||
                (k >= 16 && k <= 20) ||
                (k >= 44 && k <= 46);
        },

        fromCharCode : function(code) {
            for (var o in this.keys) {
                if (this.keys[o] === code) return o;
            }
        }
    },
